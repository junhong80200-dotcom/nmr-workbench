<script>
    // ---------- 模块1：下拉选择不良描述 ----------
    const descSelect = document.getElementById('descSelect');
    const applyDescBtn = document.getElementById('applyDescBtn');
    const defectDescTextarea = document.getElementById('defectDesc');
    applyDescBtn.addEventListener('click', () => {
        const selected = descSelect.value;
        if (selected) {
            defectDescTextarea.value = selected;
            updateAgileText();
        } else {
            alert('请先选择一个常见描述');
        }
    });

    // 生成 Agile 文本
    const generateBtn = document.getElementById('generateBtn');
    const copyResultBtn = document.getElementById('copyResultBtn');
    const agileResultDiv = document.getElementById('agileResult');
    function updateAgileText() {
        const lot = document.getElementById('lotQty').value.trim() || '?';
        const reject = document.getElementById('rejectQty').value.trim() || '0';
        const desc = defectDescTextarea.value.trim() || '无描述';
        const formatted = `Lot Qty: ${lot} | Reject Qty: ${reject} | Non-conforming: ${desc}`;
        agileResultDiv.innerText = formatted;
        return formatted;
    }
    generateBtn.addEventListener('click', updateAgileText);
    copyResultBtn.addEventListener('click', () => {
        const text = agileResultDiv.innerText;
        navigator.clipboard.writeText(text).then(() => alert('已复制到剪贴板')).catch(() => alert('复制失败，手动复制'));
    });
    updateAgileText();

    // ---------- 模块2：库存位置 (localStorage) ----------
    const STORAGE_KEY = 'nmr_location_map';
    function loadLocationMap() {
        const raw = localStorage.getItem(STORAGE_KEY);
        if (raw) return JSON.parse(raw);
        return { "PN12345": "A3-2排", "NMR-001": "B1-05架" };
    }
    function saveLocationMap(map) {
        localStorage.setItem(STORAGE_KEY, JSON.stringify(map));
    }
    let locationMap = loadLocationMap();

    function renderLocationList() {
        const container = document.getElementById('locationList');
        const entries = Object.entries(locationMap);
        if (entries.length === 0) {
            container.innerHTML = '<div style="color:gray;">暂无记录，请添加物料号和位置</div>';
            return;
        }
        let html = '';
        for (const [part, loc] of entries) {
            html += `<div class="loc-item">
                        <div><span class="loc-part">${escapeHtml(part)}</span> → <span class="loc-loc">${escapeHtml(loc)}</span></div>
                        <button class="delete-btn" data-part="${escapeHtml(part)}">删除</button>
                     </div>`;
        }
        container.innerHTML = html;
        document.querySelectorAll('.delete-btn').forEach(btn => {
            btn.addEventListener('click', (e) => {
                const part = btn.getAttribute('data-part');
                if (confirm(`删除物料 ${part} 的位置记录？`)) {
                    delete locationMap[part];
                    saveLocationMap(locationMap);
                    renderLocationList();
                    const curPart = document.getElementById('partNo').value.trim();
                    if (curPart === part) document.getElementById('queryResult').innerHTML = '已删除该物料记录';
                }
            });
        });
    }

    function escapeHtml(str) { return str.replace(/[&<>]/g, function(m){if(m==='&') return '&amp;'; if(m==='<') return '&lt;'; if(m==='>') return '&gt;'; return m;}); }

    document.getElementById('queryLocationBtn').addEventListener('click', () => {
        const part = document.getElementById('partNo').value.trim();
        const resultDiv = document.getElementById('queryResult');
        if (!part) { resultDiv.innerHTML = '❓ 请输入物料号'; return; }
        const loc = locationMap[part];
        if (loc) {
            resultDiv.innerHTML = `📍 物料 <strong>${escapeHtml(part)}</strong> 的存储位置：<span style="color:#2c7da0;">${escapeHtml(loc)}</span><br>可直接去该位置取料。`;
            document.getElementById('newLocation').value = loc;
        } else {
            resultDiv.innerHTML = `⚠️ 未找到物料 ${escapeHtml(part)} 的位置记录，请在右侧填写位置并保存。`;
        }
    });

    document.getElementById('saveLocationBtn').addEventListener('click', () => {
        const part = document.getElementById('partNo').value.trim();
        const loc = document.getElementById('newLocation').value.trim();
        if (!part || !loc) { alert('物料号和位置都不能为空'); return; }
        locationMap[part] = loc;
        saveLocationMap(locationMap);
        renderLocationList();
        document.getElementById('queryResult').innerHTML = `✅ 已保存：${escapeHtml(part)} → ${escapeHtml(loc)}`;
        alert('保存成功');
    });

    document.getElementById('exportLocationsBtn').addEventListener('click', () => {
        let csvContent = "物料号,存储位置\n";
        for (const [part, loc] of Object.entries(locationMap)) {
            csvContent += `"${part}","${loc}"\n`;
        }
        const blob = new Blob(["\uFEFF" + csvContent], { type: "text/csv;charset=utf-8;" });
        const link = document.createElement('a');
        const url = URL.createObjectURL(blob);
        link.href = url;
        link.setAttribute("download", "nmr_locations.csv");
        document.body.appendChild(link);
        link.click();
        document.body.removeChild(link);
        URL.revokeObjectURL(url);
    });

    renderLocationList();

    // ---------- 模块3：下载批处理脚本 ----------
    document.getElementById('downloadBatBtn').addEventListener('click', () => {
        const batContent = `@echo off
setlocal enabledelayedexpansion
echo 批量重命名并压缩照片（需安装 ImageMagick）
set /p jobno=请输入工单号（例如 NMR001）:
set count=1
for %%f in (*.jpg *.png *.jpeg) do (
    echo 处理 %%f
    magick "%%f" -resize 800x800 "%jobno%_!count!.jpg"
    set /a count+=1
)
echo 完成！共处理 %count% 个文件。
pause
`;
        const blob = new Blob([batContent], { type: "text/plain" });
        const link = document.createElement('a');
        const url = URL.createObjectURL(blob);
        link.href = url;
        link.setAttribute("download", "rename_resize_photos.bat");
        document.body.appendChild(link);
        link.click();
        document.body.removeChild(link);
        URL.revokeObjectURL(url);
    });
</script>
